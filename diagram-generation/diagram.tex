\documentclass[tikz,border=2pt]{standalone}
\usepackage{tikz}
\begin{document}
\begin{tikzpicture}
  % Axes
  \draw[->] (-1.5,0) -- (5.5,0) node[right] {$x$};
  \draw[->] (0,-1) -- (0,6) node[above] {$y$};
  
  % Ticks and labels on x-axis
  \foreach \x in {-1, 0, 1, 2, 3, 4, 5}
    \draw (\x,0.1) -- (\x,-0.1) node[below] {\x};
  
  % Ticks and labels on y-axis
  \foreach \y in {1, 2, 3, 4, 5}
    \draw (0.1,\y) -- (-0.1,\y) node[left] {\y};
  
  % Parabola
  \draw[domain=-1:5, smooth, variable=\x] plot ({\x},{\x*\x - 4*\x + 3});
  
  % Turning point
  \filldraw (2,-1) circle (2pt) node[above right] {Turning Point $(2, -1)$};
  
  % Additional points
  \filldraw (1,0) circle (1pt) node[below right] {$(1, 0)$};
  \filldraw (3,0) circle (1pt) node[below left] {$(3, 0)$};
\end{tikzpicture}
\end{document}
