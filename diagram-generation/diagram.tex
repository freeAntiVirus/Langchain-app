\documentclass[tikz,border=2pt]{standalone}
\usepackage{tikz}
\begin{document}
\begin{tikzpicture}
  % Draw the axes
  \draw[->] (-0.5,0) -- (4.5,0) node[right] {$x$};
  \draw[->] (0,-2.5) -- (0,3.5) node[above] {$y$};
  
  % Draw the function curve
  \draw[domain=0:4,smooth,variable=\x] plot ({\x},{\x*\x*\x - 3*\x*\x + 2*\x});
  
  % Label the curve
  \node at (3.5,2.5) {$y = f(x)$};
  
  % Mark and label the stationary points
  \filldraw (0,0) circle (2pt) node[below left] {$(0,0)$};
  \filldraw (1,0) circle (2pt) node[below right] {$(1,0)$};
  \filldraw (2,0) circle (2pt) node[below right] {$(2,0)$};
  
  % Add ticks on the x-axis
  \foreach \x in {1,2,3,4}
    \draw (\x,0.1) -- (\x,-0.1) node[below] {\x};
  
  % Add ticks on the y-axis
  \foreach \y in {-2,-1,1,2,3}
    \draw (0.1,\y) -- (-0.1,\y) node[left] {\y};
\end{tikzpicture}
\end{document}
